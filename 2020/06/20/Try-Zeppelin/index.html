%%bash
cat << EOF | sudo tee /opt/tljh/state/dynamic/zeppelin.toml > /dev/null

[frontends.zeppelin]
backend = "zeppelin"
passHostHeader = true

[frontends.zeppelin.routes.f1]
rule = "PathPrefixStrip:/zeppelin/"

[backends.zeppelin.servers.s1]
url = "http://127.0.0.1:8080"
weight = 1
EOF
